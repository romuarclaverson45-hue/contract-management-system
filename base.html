document.addEventListener('DOMContentLoaded', () => { const searchInput = document.getElementById('search-input'); const searchBtn = document.getElementById('search-btn'); const clearBtn = document.getElementById('clear-btn'); const resultCount = document.getElementById('result-count'); const cardsView = document.getElementById('cards-view'); const tableBody = document.getElementById('table-body'); const emptyState = document.getElementById('empty-state'); const cardsViewWrap = document.getElementById('cards-view'); const tableViewWrap = document.getElementById('table-view'); const viewCardsBtn = document.getElementById('view-cards'); const viewTableBtn = document.getElementById('view-table'); function escapeHtml(str) { if (str === null || str === undefined) return ''; return String(str) .replace(/&/g, '&') .replace(//g, '>'); } function money(v) { const n = parseFloat(v) || 0; return '₱' + n.toFixed(2); } function renderCards(results) { cardsView.innerHTML = results.map(r => `
${r.photo_url ? `` : `
No Photo
` }
${escapeHtml(r.contract_name)}
${escapeHtml(r.contract_id)}
Contract Amount${money(r.contract_amount)}
Premium${money(r.premium)}
Others${money(r.others)}
Total${money(r.total_amount)}
✉️ ${escapeHtml(r.email) || '—'}
📞 ${escapeHtml(r.phone) || '—'}
🗓️ ${escapeHtml(r.created_at)}
`).join(''); } function renderTable(results) { tableBody.innerHTML = results.map(r => ` ${r.photo_url ? `` : `
N/A
` } ${escapeHtml(r.contract_id)} ${escapeHtml(r.contract_name)} ${money(r.contract_amount)} ${money(r.premium)} ${money(r.others)} ${money(r.total_amount)} ${escapeHtml(r.email)} ${escapeHtml(r.phone)} ${escapeHtml(r.created_at)} `).join(''); } async function attachDeleteHandlers() { document.querySelectorAll('.delete-btn').forEach(btn => { btn.addEventListener('click', async () => { if (!confirm('Delete this record? This cannot be undone.')) return; const id = btn.dataset.id; const res = await fetch(`/api/contracts/${id}`, { method: 'DELETE' }); const data = await res.json(); if (data.success) { runSearch(); } else { alert(data.message || 'Failed to delete record.'); } }); }); } async function runSearch() { const q = searchInput.value.trim(); resultCount.textContent = 'Searching...'; try { const res = await fetch(`/api/search?q=${encodeURIComponent(q)}`); const data = await res.json(); const results = data.results || []; resultCount.textContent = `${results.length} record(s) found`; if (results.length === 0) { cardsView.innerHTML = ''; tableBody.innerHTML = ''; emptyState.classList.remove('hidden'); } else { emptyState.classList.add('hidden'); renderCards(results); renderTable(results); attachDeleteHandlers(); } } catch (err) { resultCount.textContent = 'Error loading records.'; } } searchBtn.addEventListener('click', runSearch); searchInput.addEventListener('keyup', (e) => { if (e.key === 'Enter') runSearch(); }); clearBtn.addEventListener('click', () => { searchInput.value = ''; runSearch(); }); viewCardsBtn.addEventListener('click', () => { cardsViewWrap.classList.remove('hidden'); tableViewWrap.classList.add('hidden'); viewCardsBtn.classList.add('active-view'); viewTableBtn.classList.remove('active-view'); }); viewTableBtn.addEventListener('click', () => { tableViewWrap.classList.remove('hidden'); cardsViewWrap.classList.add('hidden'); viewTableBtn.classList.add('active-view'); viewCardsBtn.classList.remove('active-view'); }); // Initial load: show all records runSearch(); });
